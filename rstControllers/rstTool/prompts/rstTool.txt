we got a working standalone cli tool to study rst controllers. We use the functional programming approach in this script.
the script is called rst.py:

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
rst.py — God-tier, MATLAB-killer CLI for RST digital control design

Purpose
-------
Design, analyze, and simulate two-degree-of-freedom RST controllers for SISO
plants in the z-domain (delay operator q := z^{-1}). Implements the full
Diophantine (Bezout) solve for R(q), S(q), and flexible choices for T(q) with
step-tracking and internal-model options.

**Directory convention (per Management)**
• Any files **generated** by this tool are **always written to `out/`**.
• Any files **consumed** by this tool are looked up under **`in/`** by default.

Key equations (notation q = z^{-1})
-----------------------------------
Plant:    A(q) y = q^{d} B(q) u + v.
Controller:  S(q) u = T(q) r - R(q) y.
Characteristic polynomial target:    A_cl(q) = A(q) S(q) + q^{d} B(q) R(q).

Features
--------
• **Square-by-construction Diophantine** with degree allocation (`--alloc S|R`).
• Degree overrides are now **safe**: if you request larger degrees than needed,
  the solver uses the minimal square core and **pads extra high-order
  coefficients with zeros** (keeps your requested orders, avoids rank issues).
• Pole specification options:
    - Direct z-plane poles (e.g., 0.6 0.6 0.3)
    - s-plane poles mapped via z = exp(s·Ts)
    - Direct A_cl(q) coefficients, or A_cl = Ac·Ao
• T(q) options:
    - `unity_dc`: unity step tracking; integrator-aware
    - `ac`: T(q) = K·Ac(q) (derive Ac = Acl/Ao if only Ao is provided)
    - `manual`: user-supplied
• Internal-model integrator: include (1 - q) factor in S-path.
• Pretty difference equations and CSV/JSON export.
• Simulation with output disturbance and noise.

Conventions
-----------
All polynomials are in q = z^{-1}: A(q) = a0 + a1 q + ··· + a_n q^n (ascending).
Strings may use spaces and/or commas. Complex numbers use Python syntax ("-3+4j").

Author: Digital Control project — MIT License
"""

from __future__ import annotations
import argparse
import json
import os
from dataclasses import dataclass
from typing import List, Tuple, Optional, Sequence, Union

import numpy as np

# ========================= Utilities: parsing & pretty =========================

def resolve_in_path(name: Optional[str]) -> Optional[str]:
    """Return path inside `in/` unless name is absolute or already inside in/."""
    if not name:
        return None
    if os.path.isabs(name):
        return name
    if name.startswith('in' + os.path.sep) or name.startswith('in/'):
        return name
    return os.path.join('in', name)


def ensure_out_path(name: Optional[str], default_basename: str) -> Optional[str]:
    """Return a path under out/ and create the parent directory."""
    if name is None or len(str(name).strip()) == 0:
        path = os.path.join('out', default_basename)
    else:
        base = os.path.basename(name)
        path = os.path.join('out', base)
    os.makedirs(os.path.dirname(path), exist_ok=True)
    return path


def parse_coeffs(s: Optional[str]) -> Optional[np.ndarray]:
    """Parse a string of numbers into a float numpy array (ascending in q)."""
    if s is None:
        return None
    toks = s.replace(',', ' ').split()
    if not toks:
        return None
    return np.array([complex(tok) for tok in toks], dtype=complex).real.astype(float)


def parse_complex_list(s: str) -> List[complex]:
    toks = s.replace(',', ' ').split()
    return [complex(t) for t in toks]


def as_array(x: Optional[Union[List[float], np.ndarray]]) -> Optional[np.ndarray]:
    if x is None:
        return None
    return np.array(x, dtype=float)


def poly_to_str(p: Sequence[float], var: str = 'q') -> str:
    terms = []
    for k, c in enumerate(p):
        if abs(c) < 1e-14:
            continue
        if k == 0:
            terms.append(f"{c:.6g}")
        elif k == 1:
            terms.append(f"{c:.6g}{var}")
        else:
            terms.append(f"{c:.6g}{var}^{k}")
    return ' + '.join(terms) if terms else '0'


def stable_all(roots: np.ndarray, tol: float = 1 - 1e-10) -> bool:
    return np.all(np.abs(roots) < tol)


# ======================= Polynomial helpers (q-domain) ========================

def poly_conv(a: np.ndarray, b: np.ndarray) -> np.ndarray:
    return np.convolve(a, b)


def poly_add(a: np.ndarray, b: np.ndarray) -> np.ndarray:
    L = max(len(a), len(b))
    aa = np.pad(a, (0, L - len(a)))
    bb = np.pad(b, (0, L - len(b)))
    return aa + bb


def poly_shift(p: np.ndarray, d: int) -> np.ndarray:
    # Multiply by q^d → shift to the right (ascending representation)
    if d <= 0:
        return p.copy()
    return np.pad(p, (d, 0))


def poly_from_z_poles(poles: Sequence[complex]) -> np.ndarray:
    out = np.array([1.0], dtype=complex)
    for z in poles:
        out = poly_conv(out, np.array([1.0, -z], dtype=complex))
    out = np.real_if_close(out, tol=1e-8)
    return out.astype(float)


def poly_from_q_factors(ac: Optional[np.ndarray], ao: Optional[np.ndarray]) -> Optional[np.ndarray]:
    if ac is None and ao is None:
        return None
    if ac is None:
        return ao.copy()
    if ao is None:
        return ac.copy()
    return poly_conv(ac, ao)


def eval_poly_at_one(p: np.ndarray) -> float:
    return float(np.sum(p))


def deconv_poly_asc(num: np.ndarray, den: np.ndarray) -> Tuple[np.ndarray, np.ndarray]:
    """Deconvolve ascending-coefficient polynomials: num ≈ den * q + r."""
    num_desc = num[::-1]
    den_desc = den[::-1]
    q_desc, r_desc = np.polydiv(num_desc, den_desc)
    q = np.real_if_close(q_desc[::-1], tol=1e-9).astype(float)
    r = np.real_if_close(r_desc[::-1], tol=1e-9).astype(float)
    # Trim tiny tails
    eps = 1e-10
    while len(q) > 1 and abs(q[-1]) < eps:
        q = q[:-1]
    while len(r) > 1 and abs(r[-1]) < eps:
        r = r[:-1]
    if len(q) == 0:
        q = np.array([0.0])
    if len(r) == 0:
        r = np.array([0.0])
    return q, r


# ========================== Degree policy & Diophantine =======================

def recommended_degrees(A: np.ndarray, B: np.ndarray, d: int, Acl: np.ndarray,
                        integrator: bool, alloc: str) -> Tuple[int, int]:
    """Minimal square-by-construction degrees (same policy as docstring)."""
    na = len(A) - 1
    nb = len(B) - 1
    Nc = len(Acl) - 1
    N0 = na + d + nb - 1
    m = max(0, Nc - N0)
    if alloc.lower() == 'r':
        degS = max(0, d + nb - 1)
        degR = max(0, na - 1 + m)
    else:
        degS = max(0, d + nb - 1 + m)
        degR = max(0, na - 1)
    # Note: 'integrator' affects parameterization (S = (1-q) Ŝ) not minimal order itself.
    return degS, degR


def build_conv_matrix(base: np.ndarray, out_len: int, shift: int, cols: int) -> np.ndarray:
    nb = len(base)
    M = np.zeros((out_len, cols))
    for j in range(cols):
        for i in range(nb):
            k = i + j + shift
            if 0 <= k < out_len:
                M[k, j] = base[i]
    return M


def ensure_rows_cover(Acl: np.ndarray, base_A: np.ndarray, B: np.ndarray, d: int,
                      degS_hat: int, degR: int) -> np.ndarray:
    """Pad Acl with zeros only as needed to cover the highest nonzero row index
    created by the current convolution layout. DO NOT pad just to match #unknowns
    (that would create zero rows and rank deficiency)."""
    highest_MA_row = degS_hat + (len(base_A) - 1)
    highest_MB_row = d + degR
    M_needed = max(len(Acl), highest_MA_row + 1, highest_MB_row + 1)
    if len(Acl) < M_needed:
        return np.pad(Acl, (0, M_needed - len(Acl)))
    return Acl


def diophantine_solve(A: np.ndarray, B: np.ndarray, d: int, Acl: np.ndarray,
                       degS: Optional[int] = None, degR: Optional[int] = None,
                       integrator: bool = False, alloc: str = 'S') -> Tuple[np.ndarray, np.ndarray]:
    """Solve A(q) S(q) + q^d B(q) R(q) = A_cl(q).

    Degree overrides are honored safely:
      - If user requests larger degrees than the minimal square ones, we solve
        with the minimal square core and pad extra high-order coefficients with
        zeros to match the requested orders. This avoids underdetermined / rank
        issues while keeping the requested polynomial orders.
    """
    A = np.array(A, dtype=float)
    B = np.array(B, dtype=float)
    Acl = np.array(Acl, dtype=float)

    # Normalize A to monic
    if abs(A[0] - 1.0) > 1e-12:
        Acl = Acl / A[0]
        B = B / A[0]
        A = A / A[0]

    # Recommended minimal degrees (square-by-construction)
    degS_rec, degR_rec = recommended_degrees(A, B, d, Acl, integrator, alloc)

    # Target (requested) degrees
    degS_tar = degS_rec if degS is None else max(0, int(degS))
    degR_tar = degR_rec if degR is None else max(0, int(degR))

    # Start effective degrees from targets, then shrink extras if underdetermined
    degS_eff = degS_tar
    degR_eff = degR_tar

    # Integrator parametrization
    S_fac = np.array([1.0, -1.0]) if integrator else np.array([1.0])
    base_A = poly_conv(A, S_fac)

    def n_unknowns(dS: int, dR: int) -> int:
        dS_hat = max(0, dS - (1 if integrator else 0))
        return (dS_hat + 1) + (dR + 1)

    # Ensure we have enough rows to cover convolution, but don't add zero rows
    Acl_eff = ensure_rows_cover(Acl, base_A, B, d,
                                degS_hat=max(0, degS_eff - (1 if integrator else 0)),
                                degR=degR_eff)

    # If unknowns exceed rows, *shrink only the extras above the recommended*.
    while n_unknowns(degS_eff, degR_eff) > len(Acl_eff):
        if degS_eff > degS_rec:
            degS_eff -= 1
        elif degR_eff > degR_rec:
            degR_eff -= 1
        else:
            # Fallback: prefer shrinking S
            if degS_eff > 0:
                degS_eff -= 1
            elif degR_eff > 0:
                degR_eff -= 1
            else:
                break
        Acl_eff = ensure_rows_cover(Acl, base_A, B, d,
                                    degS_hat=max(0, degS_eff - (1 if integrator else 0)),
                                    degR=degR_eff)

    # Build final linear system with effective degrees
    degS_hat = max(0, degS_eff - (1 if integrator else 0))
    nS_unknowns = degS_hat + 1
    nR_unknowns = degR_eff + 1

    MA = build_conv_matrix(base_A, out_len=len(Acl_eff), shift=0, cols=nS_unknowns)
    MB = build_conv_matrix(B, out_len=len(Acl_eff), shift=d, cols=nR_unknowns)
    Phi = np.block([MA, MB])
    bvec = Acl_eff.copy()

    # Solve; if rank-deficient (rare), try shrinking the remaining extras;
    # finally fall back to min-norm pseudoinverse.
    for _ in range(10):
        x, _, rank, _ = np.linalg.lstsq(Phi, bvec, rcond=None)
        if rank == Phi.shape[1]:
            break
        # Try to shrink remaining extras if any (prefer S extras, then R)
        shrunk = False
        if degS_eff > degS_rec and nS_unknowns > 1:
            degS_eff -= 1
            degS_hat = max(0, degS_eff - (1 if integrator else 0))
            nS_unknowns = degS_hat + 1
            MA = build_conv_matrix(base_A, out_len=len(Acl_eff), shift=0, cols=nS_unknowns)
            Phi = np.block([MA, MB])
            shrunk = True
        elif degR_eff > degR_rec and nR_unknowns > 1:
            degR_eff -= 1
            nR_unknowns = degR_eff + 1
            MB = build_conv_matrix(B, out_len=len(Acl_eff), shift=d, cols=nR_unknowns)
            Phi = np.block([MA, MB])
            shrunk = True
        if not shrunk:
            # Pseudoinverse fallback to avoid hard failure
            x = np.linalg.pinv(Phi) @ bvec
            rank = Phi.shape[1]
            break

    S_hat = x[:nS_unknowns]
    R_eff = x[nS_unknowns:]
    S_eff = poly_conv(S_fac, S_hat)

    # Pad with zeros to match the *requested* target degrees.
    if len(S_eff) < (degS_tar + 1):
        S_eff = np.pad(S_eff, (0, degS_tar + 1 - len(S_eff)))
    if len(R_eff) < (degR_tar + 1):
        R_eff = np.pad(R_eff, (0, degR_tar + 1 - len(R_eff)))

    S_eff = np.real_if_close(S_eff, tol=1e-9).astype(float)
    R_eff = np.real_if_close(R_eff, tol=1e-9).astype(float)
    return S_eff, R_eff


# ========================= T-selection (with integrator) ======================

def choose_T(mode: str,
             B: np.ndarray,
             Acl: np.ndarray,
             Ac: Optional[np.ndarray],
             Ao: Optional[np.ndarray],
             T_manual: Optional[np.ndarray],
             integrator: bool,
             R: np.ndarray) -> np.ndarray:
    """Select T(q) with correct unity-DC handling and integrator logic.

    unity_dc:
      - no integrator → T = [A_cl(1)/B(1)]
      - with integrator → T = [R(1)] (to enforce step tracking)
    ac:
      - use Ac if provided; else derive Ac := Acl/Ao when possible
      - no integrator → K = A_cl(1) / (B(1)·Ac(1)), T = K·Ac
      - with integrator → scale so (K·Ac)(1) = R(1)
    manual: return given T.
    """
    mode = mode.lower()
    B1 = eval_poly_at_one(B)
    Acl1 = eval_poly_at_one(Acl)
    R1 = eval_poly_at_one(R)

    if mode == 'unity_dc':
        if integrator:
            return np.array([R1], dtype=float)
        if abs(B1) < 1e-12:
            return np.array([1.0], dtype=float)
        KT = Acl1 / B1
        return np.array([KT], dtype=float)

    if mode == 'ac':
        Ac_use = Ac
        if Ac_use is None and Ao is not None:
            q, r = deconv_poly_asc(Acl, Ao)
            if np.linalg.norm(r) < 1e-8:
                Ac_use = q
        if Ac_use is None:
            raise ValueError("T mode 'ac' requires Ac(q), or Acl/Ao so Ac can be derived.")
        Ac1 = eval_poly_at_one(Ac_use)
        if integrator:
            if abs(Ac1) < 1e-12:
                K = 1.0
            else:
                K = R1 / Ac1
        else:
            denom = B1 * Ac1
            K = 1.0 if abs(denom) < 1e-12 else (Acl1 / denom)
        return (K * Ac_use).astype(float)

    if mode == 'manual':
        if T_manual is None:
            raise ValueError("T mode 'manual' requires --T coefficients.")
        return np.array(T_manual, dtype=float)

    raise ValueError("Unknown T mode. Use 'unity_dc', 'ac', or 'manual'.")


# ================================= Simulation =================================
@dataclass
class SimResult:
    t: np.ndarray
    r: np.ndarray
    y: np.ndarray
    u: np.ndarray
    v: np.ndarray


def simulate_closed_loop(A: np.ndarray, B: np.ndarray, d: int,
                          R: np.ndarray, S: np.ndarray, Tq: np.ndarray,
                          N: int = 200, r_step: float = 1.0,
                          v_step: float = 0.0, v_k0: int = 0,
                          noise_sigma: float = 0.0,
                          seed: Optional[int] = 0) -> SimResult:
    """Closed-loop simulation with **causal scheduling**.

    d ≥ 1: update y[k] first using u[k-d], then compute u[k] from current y[k].
    d = 0: solve 2×2 algebraic loop for (u[k], y[k]).
    """
    rng = np.random.default_rng(seed)

    # Normalize A to monic
    if abs(A[0] - 1.0) > 1e-12:
        B = B / A[0]
        A = A / A[0]

    na = len(A) - 1
    nb = len(B) - 1
    mS = len(S) - 1
    mR = len(R) - 1
    mT = len(Tq) - 1

    y = np.zeros(N)
    u = np.zeros(N)
    r = np.zeros(N)
    v = np.zeros(N)

    r[:] = r_step
    if v_step != 0.0 and 0 <= v_k0 < N:
        v[v_k0:] = v_step

    def get(arr: np.ndarray, k: int) -> float:
        return 0.0 if k < 0 else arr[k]

    for k in range(N):
        noise = rng.normal(0.0, noise_sigma) if noise_sigma > 0 else 0.0

        if d >= 1:
            # Plant update first (uses u up to k-d)
            Bu = 0.0
            for j in range(0, nb + 1):
                Bu += B[j] * get(u, k - d - j)
            Ay = 0.0
            for i in range(1, na + 1):
                Ay += A[i] * get(y, k - i)
            y[k] = Bu + v[k] + noise - Ay

            # Controller uses current y[k]
            Su = 0.0
            for i in range(1, mS + 1):
                Su += S[i] * get(u, k - i)
            Tr = 0.0
            for i in range(0, mT + 1):
                Tr += Tq[i] * get(r, k - i)
            Ry = 0.0
            for i in range(0, mR + 1):
                Ry += R[i] * get(y, k - i)
            u[k] = (Tr - Ry - Su) / S[0]

        else:  # d == 0 → algebraic loop between u[k] and y[k]
            # Prepare history sums
            Su_hist = 0.0
            for i in range(1, mS + 1):
                Su_hist += S[i] * get(u, k - i)
            Tr = 0.0
            for i in range(0, mT + 1):
                Tr += Tq[i] * get(r, k - i)
            Ry_hist = 0.0
            for i in range(1, mR + 1):
                Ry_hist += R[i] * get(y, k - i)
            c1 = Tr - Su_hist - Ry_hist  # S0*u + R0*y

            Ay_hist = 0.0
            for i in range(1, na + 1):
                Ay_hist += A[i] * get(y, k - i)
            Bu_hist = 0.0
            for j in range(1, nb + 1):
                Bu_hist += B[j] * get(u, k - j)
            c2 = -Ay_hist + Bu_hist + v[k] + noise  # y - B0*u

            S0 = S[0]
            R0 = R[0] if mR >= 0 else 0.0
            B0 = B[0] if nb >= 0 else 0.0

            # Solve [[S0, R0], [-B0, 1]] [u, y] = [c1, c2]
            det = S0*1.0 + B0*R0
            if abs(det) < 1e-12:
                det = det + 1e-9
            u[k] = (c1*1.0 - R0*c2) / det
            y[k] = (S0*c2 + B0*c1) / det

    t = np.arange(N)
    return SimResult(t=t, r=r, y=y, u=u, v=v)


# ================================ I/O helpers =================================

def save_csv(sim: SimResult, path: str) -> None:
    import csv
    with open(path, 'w', newline='') as f:
        w = csv.writer(f)
        w.writerow(["k", "r", "y", "u", "v"])
        for k in range(len(sim.t)):
            w.writerow([int(sim.t[k]), float(sim.r[k]), float(sim.y[k]), float(sim.u[k]), float(sim.v[k])])


def save_json_design(path: str, payload: dict) -> None:
    with open(path, 'w') as f:
        json.dump(payload, f, indent=2)


def diff_eq_strings(R: np.ndarray, S: np.ndarray, Tq: np.ndarray) -> Tuple[str, str]:
    def poly_rhs_str(name: str, p: np.ndarray) -> str:
        terms = []
        for i, c in enumerate(p):
            if abs(c) < 1e-14:
                continue
            if i == 0:
                terms.append(f"{c:.6g} {name}[k]")
            else:
                terms.append(f"{c:.6g} {name}[k-{i}]")
        return ' + '.join(terms) if terms else '0'

    left = poly_rhs_str('u', S)
    right_T = poly_rhs_str('r', Tq)
    right_R = poly_rhs_str('y', R)

    ctrl_eq = f"{left} = ({right_T}) - ({right_R})"

    # Explicit u[k]
    num_terms = []
    if len(Tq) > 0:
        for i, c in enumerate(Tq):
            if abs(c) >= 1e-14:
                idx = '' if i == 0 else f"-{i}"
                num_terms.append(f"{c:.6g} r[k{idx}]")
    if len(R) > 0:
        for i, c in enumerate(R):
            if abs(c) >= 1e-14:
                idx = '' if i == 0 else f"-{i}"
                num_terms.append(f"{-c:.6g} y[k{idx}]")
    if len(S) > 1:
        for i, c in enumerate(S[1:], start=1):
            if abs(c) >= 1e-14:
                num_terms.append(f"{-c:.6g} u[k-{i}]")
    denom = S[0]
    ueq = "u[k] = (" + ' + '.join(num_terms) + f") / {denom:.6g}"

    return ctrl_eq, ueq


# ================================ Main CLI ====================================

def main():
    p = argparse.ArgumentParser(
        description=(
            "RST digital control design & simulation (q = z^{-1}).\n"
            "Outputs forced to out/; JSON specs loaded from in/ by default."
        ),
        formatter_class=argparse.ArgumentDefaultsHelpFormatter,
    )

    # ---- Plant ----
    p.add_argument('--A', help='A(q) coefficients (q = z^{-1}), e.g. "1 -0.8"')
    p.add_argument('--B', help='B(q) coefficients, e.g. "0.5"')
    p.add_argument('--d', type=int, default=None, help='Input delay (integer ≥ 0)')
    p.add_argument('--Ts', type=float, default=None, help='Sampling period for s→z mapping')

    # ---- Target closed-loop ----
    g = p.add_mutually_exclusive_group(required=False)
    g.add_argument('--Acl', help='Target A_cl(q) coefficients directly')
    g.add_argument('--poles', help='Target z-plane poles (space/comma separated)')
    g.add_argument('--spoles', help='Target s-plane poles; mapped with z=exp(s*Ts)')

    # Optional Ac/Ao factorization
    p.add_argument('--Ac', help='Ac(q) coefficients (dominant command dynamics)')
    p.add_argument('--Ao', help='Ao(q) coefficients (noise/observer extra poles)')

    # Degrees & integrator
    p.add_argument('--degS', type=int, help='Degree of S(q) (overrides auto)')
    p.add_argument('--degR', type=int, help='Degree of R(q) (overrides auto)')
    p.add_argument('--alloc', choices=['S','R'], default='S', help='Where to allocate extra degree (beyond minimal): S or R')
    p.add_argument('--integrator', action='store_true', help='Include (1 - q) factor in S-path')

    # Prefilter T
    p.add_argument('--Tmode', default=None, choices=['unity_dc', 'ac', 'manual'], help='T(q) selection mode')
    p.add_argument('--T', help='Manual T(q) coefficients if --Tmode manual')

    # Simulation
    p.add_argument('--step', type=int, default=None, help='Sim length N (samples) for step test')
    p.add_argument('--r_step', type=float, default=None, help='Reference step amplitude')
    p.add_argument('--v_step', type=float, default=None, help='Output disturbance step amplitude')
    p.add_argument('--v_k0', type=int, default=None, help='Disturbance step start index k0')
