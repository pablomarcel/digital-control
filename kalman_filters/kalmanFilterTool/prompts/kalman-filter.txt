we got a working standalone cli tool to study kalman filters. We use the functional programming approach in this script.
the script is called kalman_filter.py:

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
kalman_filter.py — Discrete Kalman filter (from first principles)

Defaults to a constant-velocity (CV) model with sample time dt:
    x = [position, velocity]^T
    A = [[1, dt], [0, 1]]
    B = [[0.5*dt^2], [dt]]  (known accel input)
    C = [1, 0]              (measure position)
    Process noise: white accel with intensity q -> Qd(dt)  (STATE-FORM)
    Measurement noise variance r -> R = [[r]]

You can override A,B,C,G,Q,R via CLI.

Key upgrades
------------
• Robust shape handling:
  - If G has shape (1,n) it is auto-transposed to (n,1).
  - R scalar -> expands to p×p * scalar when p>1.
  - Q can be STATE-FORM (n×n) or INPUT-FORM (m×m with G∈R^{n×m}).
• Internals use:
  - GQGT = Q_state (if state-form) OR (G @ Qw @ G.T) (if input-form),
  - Simulation uses the same noise model (v ~ N(0,Q_state) OR w ~ N(0,Qw)) — no mismatch.
• Directory policy: relative outputs saved under --out-dir (default: ./out).

Examples
--------
1) CV model, time-varying KF, Plotly+MPL, save all:
   python kalman_filter.py --dt 0.05 --T 10 --q 0.25 --r 4.0 \
     --backend both --save_csv kf_cv.csv --save_png kf_cv.png --save_html kf_cv.html

2) CV model, steady-state KF:
   python kalman_filter.py --dt 0.05 --T 10 --q 0.1 --r 9.0 --steady \
     --backend mpl --save_png kf_cv_ss.png

3) Custom model override (ROW G auto-transposed):
   python kalman_filter.py \
     --A "1 0.05; 0 1" \
     --B "0.00125; 0.05" \
     --C "1 0" \
     --G "0 1" \
     --Q "0.01 0; 0 0.1" \
     --R "4" \
     --backend none --save_csv kf_custom.csv
"""
from __future__ import annotations
import argparse, os, json
import numpy as np

# ----------------------- directory utilities ----------------------- #

def ensure_dir(path: str) -> None:
    d = os.path.dirname(path)
    if d and not os.path.exists(d):
        os.makedirs(d, exist_ok=True)

def out_path(out_dir: str, name: str | None, default_name: str) -> str:
    """
    If 'name' is absolute or has a directory part, use as-is.
    Otherwise, write under out_dir.
    """
    base = os.path.basename(name) if name else default_name
    if name and (os.path.isabs(name) or os.path.dirname(name)):
        ensure_dir(name)
        return name
    p = os.path.join(out_dir, base)
    ensure_dir(p)
    return p

# ----------------------- text-to-matrix utilities ----------------------- #

def parse_matrix(s: str) -> np.ndarray:
    """
    Parse a matrix like "1 2; 3 4" or "1,2; 3,4".
    Accepts complex tokens; drops tiny imaginary parts safely.
    """
    rows = [r.strip() for r in s.strip().split(";") if r.strip()]
    mat = [[complex(x.replace("i","j")) for x in r.replace(",", " ").split()] for r in rows]
    arr = np.array(mat, dtype=complex)
    arr = np.real_if_close(arr, tol=1e8)
    if np.iscomplexobj(arr):
        if np.max(np.abs(arr.imag)) > 1e-12:
            raise SystemExit("[Fatal] Matrix has non-negligible imaginary parts.")
        arr = arr.real
    return arr.astype(float)

def parse_vector(s: str) -> np.ndarray:
    arr = np.array([complex(x.replace("i","j")) for x in s.replace(",", " ").split()], dtype=complex)
    arr = np.real_if_close(arr, tol=1e8)
    if np.iscomplexobj(arr) and np.max(np.abs(arr.imag)) > 1e-12:
        raise SystemExit("[Fatal] Vector has non-negligible imaginary parts.")
    return np.asarray(arr.real, dtype=float).reshape(-1, 1)

# ----------------------- shape helpers ----------------------- #

def coerce_shapes(A, B, C, G, Q, R, q_default, dt, args):
    """
    Returns (A,B,C,G,Q_state, Qw, R, GQGT, noise_mode)
    noise_mode: 'state' uses v~N(0,Q_state) ; 'input' uses w~N(0,Qw) with x+=G w
    """
    n = A.shape[0]

    # ---- C shape ----
    if C.ndim == 1:
        C = C.reshape(1, -1)
    assert C.shape[1] == n, f"C must have {n} columns (got {C.shape})"
    p = C.shape[0]

    # ---- B shape (allow column vector) ----
    if B.ndim == 1:
        B = B.reshape(-1, 1)
    assert B.shape[0] == n, f"B must have {n} rows (got {B.shape})"

    # ---- R shape: scalar -> p×p * scalar ; 1×1 -> expand when p>1 ----
    if R.ndim == 0:
        R = np.array([[float(R)]])
    if R.shape == (1, 1) and p > 1:
        R = np.eye(p) * float(R[0, 0])
    if R.shape == (p,):
        R = R.reshape(p, 1)
    if R.shape == (p, 1):
        R = np.diagflat(R)
    assert R.shape == (p, p), f"R must be {p}x{p} (got {R.shape})"

    # ---- Decide noise mode & build GQGT consistently ----
    # Default CV model (when args.Q is None): we created a STATE-FORM Q (n×n).
    # If user passed a Q:
    #   * If Q is n×n   -> STATE-FORM noise v ~ N(0,Q), ignore G in covariance (but allow G for input-form if user also gives input-form Q).
    #   * If Q is m×m and G is n×m -> INPUT-FORM noise w ~ N(0,Qw).
    #   * If Q is scalar -> interpret as m×m * scalar if G given; otherwise n×n * scalar.
    Q_state = None
    Qw = None
    noise_mode = 'state'  # default

    # Coerce G if provided
    if G is not None:
        # If row vector (1×n), transpose to column (n×1)
        if G.shape[0] != n and G.shape[1] == n:
            G = G.T
        assert G.shape[0] == n, f"G must have {n} rows (got {G.shape})"

    # Interpret Q
    if args.Q is None:
        # We already computed Q as state-form CV discretization
        Q_state = Q
        noise_mode = 'state'
    else:
        # User-specified Q
        if Q.ndim == 0:
            # scalar: if G known -> input-form; else state-form
            if G is not None:
                m = G.shape[1]
                Qw = float(Q) * np.eye(m)
                noise_mode = 'input'
            else:
                Q_state = float(Q) * np.eye(n)
                noise_mode = 'state'
        elif Q.shape == (n, n):
            Q_state = Q
            noise_mode = 'state'
        elif G is not None and Q.shape == (G.shape[1], G.shape[1]):
            Qw = Q
            noise_mode = 'input'
        else:
            raise SystemExit(f"[Fatal] Could not interpret Q with shapes: Q={Q.shape}, G={(None if G is None else G.shape)}, A={A.shape}")

    # Build GQGT used in Riccati and time update
    if noise_mode == 'state':
        GQGT = Q_state
        if G is None:
            # Provide a dummy G (unused in state mode)
            G = np.zeros((n, 0))
    else:
        GQGT = G @ Qw @ G.T

    return A, B, C, G, Q_state, Qw, R, GQGT, noise_mode

# ----------------------- DARE (steady-state) ----------------------- #

def dare_estimation(A: np.ndarray, C: np.ndarray, GQGT: np.ndarray, R: np.ndarray) -> np.ndarray:
    """
    Solve the discrete-time algebraic Riccati equation for estimation:
       P = A P A' - A P C' (C P C' + R)^-1 C P A' + G Q G'
    Prefer SciPy; otherwise do a simple fixed-point iteration.
    """
    try:
        from scipy.linalg import solve_discrete_are
        P = solve_discrete_are(A.T, C.T, GQGT, R)
        return P
    except Exception:
        # Iterative fallback (robust for small n)
        P = np.eye(A.shape[0])
        for _ in range(10000):
            S = C @ P @ C.T + R
            K = P @ C.T @ np.linalg.inv(S)
            Pn = A @ P @ A.T - A @ P @ C.T @ np.linalg.inv(S) @ C @ P @ A.T + GQGT
            if np.linalg.norm(Pn - P, ord='fro') < 1e-10:
                P = Pn
                break
            P = Pn
        return P

# ----------------------- main ----------------------- #
def main():
    ap = argparse.ArgumentParser(
        description="Discrete Kalman filter simulation (time-varying or steady-state).",
        formatter_class=argparse.ArgumentDefaultsHelpFormatter,
        allow_abbrev=False,
    )
    # High-level CV model params
    ap.add_argument("--dt", type=float, default=0.05, help="Sample time (s)")
    ap.add_argument("--T",  type=float, default=10.0, help="Simulation duration (s)")
    ap.add_argument("--q",  type=float, default=0.25, help="Accel noise intensity (CV model)")
    ap.add_argument("--r",  type=float, default=4.0,   help="Measurement noise variance (CV model)")
    ap.add_argument("--u",  type=float, default=0.0,   help="Constant input acceleration (m/s^2)")

    # Overrides
    ap.add_argument("--A", help='Override A, e.g. "1 0.05; 0 1"')
    ap.add_argument("--B", help='Override B, e.g. "0.00125; 0.05"')
    ap.add_argument("--C", help='Override C, e.g. "1 0" or "1 0; 0 1"')
    ap.add_argument("--G", help='Override G, process noise input matrix (n×m)')
    ap.add_argument("--Q", help='Override Q: n×n (state-form) or m×m (input-form with G)')
    ap.add_argument("--R", help='Override R (p×p); scalar expands to p×p*scalar')

    # Initials
    ap.add_argument("--x0",  default="0 0", help='Initial estimate x0 (column), e.g. "0 0"')
    ap.add_argument("--P0",  default="100 0; 0 100", help='Initial covariance P0')
    ap.add_argument("--xtrue0", default="0 0", help='True initial state (for sim)')

    ap.add_argument("--steady", action="store_true", help="Use steady-state KF (DARE) instead of time-varying")
    ap.add_argument("--seed", type=int, default=1, help="Random seed")
    ap.add_argument("--backend", choices=["mpl","plotly","both","none"], default="both",
                    help="Plotting backend(s)")

    # Output files (relative paths go under --out-dir)
    ap.add_argument("--save_csv",  default=None, help="CSV filename")
    ap.add_argument("--save_png",  default=None, help="PNG filename (MPL)")
    ap.add_argument("--save_html", default=None, help="HTML filename (Plotly)")
    ap.add_argument("--no_show", action="store_true", help="Do not open plot windows (MPL)")

    # Directory policy
    ap.add_argument("--in-dir",  default="in",  help="default input directory for future file inputs")
    ap.add_argument("--out-dir", default="out", help="output directory for relative filenames")
    args = ap.parse_args()

    rng = np.random.default_rng(args.seed)

    # Time axis
    N = int(np.round(args.T / args.dt)) + 1
    T = np.linspace(0.0, args.T, N)
    dt = args.dt

    # ----- Build model -----
    A = parse_matrix(args.A) if args.A is not None else np.array([[1.0, dt],[0.0, 1.0]], dtype=float)
    B = parse_matrix(args.B) if args.B is not None else np.array([[0.5*dt**2],[dt]], dtype=float)
    C = parse_matrix(args.C) if args.C is not None else np.array([[1.0, 0.0]], dtype=float)
    G = parse_matrix(args.G) if args.G is not None else np.array([[0.5*dt**2],[dt]], dtype=float)

    n = A.shape[0]

    # Default CV discrete process noise (STATE-FORM n×n)
    if args.Q is None:
        q = float(args.q)
        Q = q * np.array([[dt**4/4, dt**3/2],
                          [dt**3/2, dt**2 ]], dtype=float)
    else:
        Q = parse_matrix(args.Q)

    # Measurement covariance
    if args.R is None:
        rvar = float(args.r)
        R = np.array([[rvar]], dtype=float)
    else:
        R = parse_matrix(args.R)

    # Initials
    xh  = parse_vector(args.x0).reshape(n, 1)
    P   = parse_matrix(args.P0)
    x   = parse_vector(args.xtrue0).reshape(n, 1)

    # Sanity for A,P0
    assert A.shape == (n, n), f"A must be square (got {A.shape})"
    assert P.shape == (n, n), f"P0 must be {n}x{n} (got {P.shape})"

    # ---- Coerce shapes and build noise model ----
    A, B, C, G, Q_state, Qw, R, GQGT, noise_mode = coerce_shapes(A, B, C, G, Q, R, args.q, dt, args)

    p = C.shape[0]

    # ----- Pre-allocate logs -----
    X_true = np.zeros((n, N))
    X_hat  = np.zeros((n, N))
    Y_meas = np.zeros((p, N))

    # Optional steady-state gain
    if args.steady:
        Pss = dare_estimation(A, C, GQGT, R)
        Sss = C @ Pss @ C.T + R
        Kss = Pss @ C.T @ np.linalg.inv(Sss)
    else:
        Pss = None
        Kss = None

    # ----- Simulate -----
    for k in range(N):
        # Log current
        X_true[:, k:k+1] = x
        X_hat[:,  k:k+1] = xh

        # Measurement
        v_meas = rng.multivariate_normal(np.zeros(p), R).reshape(p, 1)
        y = C @ x + v_meas
        Y_meas[:, k:k+1] = y

        # Kalman update
        if args.steady:
            K = Kss
            nu = y - C @ xh
            xh = xh + K @ nu
            P = (np.eye(n) - K @ C) @ P  # track P roughly
        else:
            # Predict
            ucol = np.full((B.shape[1], 1), args.u, dtype=float)
            xh_pred = A @ xh + B @ ucol
            P_pred  = A @ P @ A.T + GQGT
            # Innovation
            S = C @ P_pred @ C.T + R
            K = P_pred @ C.T @ np.linalg.inv(S)
            nu = y - C @ xh_pred
            # Update
            xh = xh_pred + K @ nu
            I = np.eye(n)
            P = (I - K @ C) @ P_pred @ (I - K @ C).T + K @ R @ K.T

        # Plant propagation (ground truth) using the SAME noise model used in filter design
        ucol = np.full((B.shape[1], 1), args.u, dtype=float)
        if noise_mode == 'state':
            v_proc = rng.multivariate_normal(np.zeros(n), Q_state).reshape(n, 1)
            x = A @ x + B @ ucol + v_proc
        else:
            m = G.shape[1]
            w = rng.multivariate_normal(np.zeros(m), Qw).reshape(m, 1)
            x = A @ x + B @ ucol + G @ w

    # ------------- CSV export -------------
    if args.save_csv:
        csv_path = out_path(args.out_dir, args.save_csv, "kf.csv")
        cols = [T.reshape(-1,1)]
        head = ["t"]
        for i in range(p):
            cols.append(Y_meas[i,:].reshape(-1,1))
            head.append(f"y{i+1}")
        for i in range(n):
            cols.append(X_true[i,:].reshape(-1,1))
            head.append(f"x{i+1}_true")
        for i in range(n):
            cols.append(X_hat[i,:].reshape(-1,1))
            head.append(f"x{i+1}_hat")
        M = np.hstack(cols)
        np.savetxt(csv_path, M, delimiter=",", header=",".join(head), comments="")
        print(f"Wrote {csv_path}")

    # ------------- Plotting -------------
    if args.backend in ("mpl","both"):
        import matplotlib.pyplot as plt
        H = max(3.2, 1.6 * n)
        fig, ax = plt.subplots(1, 1, figsize=(8.0, H))
        for i in range(n):
            ax.plot(T, X_true[i,:], label=f"x{i+1} true")
            ax.plot(T, X_hat[i,:],  label=f"x{i+1} hat")
        for i in range(p):
            ax.plot(T, Y_meas[i,:], alpha=0.35, linestyle="--", label=f"y{i+1} meas")
        ax.grid(True, alpha=0.3)
        ax.set_title("Kalman filter (discrete-time)")
        ax.set_xlabel("t (s)")
        ax.set_ylabel("value")
        ax.legend(ncol=2)
        fig.tight_layout()
        if args.save_png:
            png_path = out_path(args.out_dir, args.save_png, "kf.png")
            fig.savefig(png_path, dpi=150, bbox_inches="tight")
            print(f"Wrote {png_path}")
        if not args.no_show and args.backend == "mpl":
            plt.show()

    if args.backend in ("plotly","both"):
        try:
            import plotly.graph_objects as go
        except Exception as e:
            raise SystemExit(f"[Fatal] Plotly requested but not available: {e}")
        fig = go.Figure()
        for i in range(n):
            fig.add_trace(go.Scatter(x=T, y=X_true[i,:], mode="lines", name=f"x{i+1} true"))
            fig.add_trace(go.Scatter(x=T, y=X_hat[i,:],  mode="lines", name=f"x{i+1} hat"))
        for i in range(p):
            fig.add_trace(go.Scatter(x=T, y=Y_meas[i,:], mode="lines",
                                     name=f"y{i+1} meas", opacity=0.35, line=dict(dash="dash")))
        fig.update_layout(
            title="Kalman filter (discrete-time)",
            xaxis_title="t (s)",
            yaxis_title="value",
            template="plotly_white",
            legend=dict(orientation="h", yanchor="bottom", y=1.02, xanchor="right", x=1),
            autosize=True,
        )
        html_path = out_path(args.out_dir, args.save_html, "kf.html")
        with open(html_path, "w", encoding="utf-8") as fh:
            fh.write(fig.to_html(full_html=True, include_plotlyjs="cdn",
                                 default_width="100%", default_height="70vh"))
        print(f"Wrote {html_path}")

    # Metadata
    meta = {
        "A": A.tolist(), "B": B.tolist(), "C": C.tolist(), "G": G.tolist(),
        "Q_state": (None if Q_state is None else Q_state.tolist()),
        "Qw": (None if Qw is None else Qw.tolist()),
        "R": R.tolist(), "dt": dt, "T": args.T, "steady": bool(args.steady),
        "noise_mode": noise_mode
    }
    with open(out_path(args.out_dir, "kf_meta.json", "kf_meta.json"), "w", encoding="utf-8") as fh:
        json.dump(meta, fh, indent=2)
    print(f"Wrote {out_path(args.out_dir, 'kf_meta.json', 'kf_meta.json')}")

if __name__ == "__main__":
    main()

# kalmanFilters.kalmanFilterTool — OOP Kalman Filter Tool

Run from the **repo root** (e.g., `.../digitalControl/`).
Entry point:
```bash
python -m kalmanFilters.kalmanFilterTool.cli [OPTIONS]
```

```bash
python -m kalmanFilters.kalmanFilterTool.cli   --dt 0.05 --T 10 --q 0.25 --r 4.0   --backend both --save_csv kf_cv.csv --save_png kf_cv.png --save_html kf_cv.html
```

def _ensure_prefix_dir(prefix: str | None) -> str | None:
    """
    If a save prefix is provided without a directory, put it under OUT_DIR.

def _json_default(o):
    """
    Robust JSON fallback:
      - numpy scalar -> Python scalar
      - complex -> [real, imag]
    """
    try:
        import numpy as _np
        if isinstance(o, _np.generic):
            return o.item()
    except Exception:
        pass
    if isinstance(o, complex):
        return [float(o.real), float(o.imag)]
    # Let json raise for anything else
    raise TypeError(f"Object of type {type(o).__name__} is not JSON serializable")
